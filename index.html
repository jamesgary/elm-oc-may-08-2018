Functions (Cont'd)


add : number -> number -> number
add num1 num2 =
  num1 + num2

greetings : String -> number -> String
greetings name age =
  "I'm " ++ name ++ ", and I'm " ++ toString age ++ " years old!"

-- repl
> greetings "Dorris" 100
"I'm Dorris, and I'm 100 years old!"

Lists


words : List String
words =
  [ "Hello", "Hola", "Bonjour" ]

louden : String -> String
louden word =
  word ++ "!"

-- repl
> List.map louden words
[ "Hello!", "Hola!", "Bonjour!" ] : List String

Records


type alias Point =
  { id : Int
  , x : Float
  , y : Float
  }

point = { id = 123, x = 3, y = 4 }
point = Point 123 3 4

-- repl
> point.x
3 : Float

The Elm Architecture

The Elm Architecture


main =
  { model = model,
  , view = view -- model -> html
  , update = update -- msg -> model -> model
  }

Model and View


type alias Model = Int

model : Model
model =
  0

view : Model -> Html Msg
view model =
  p [] [ text (toString model) ]

Update


type Msg
  = Increment
  | Decrement

update : Msg -> Model -> Model
update msg model =
  case msg of
    Increment ->
      model + 1

    Decrement ->
      model - 1

View (Revisited)


view : Model -> Html Msg
view model =
  div []
    [ p [] [ text (toString model) ]
    , button [ onClick Increment ] [ text "+" ]
    , button [ onClick Decrement ] [ text "-" ]
    ]
